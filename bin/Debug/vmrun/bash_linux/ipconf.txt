#!/usr/local/bin/bash 
rm -f /etc/rc.conf

echo "hostname=\"ominihost\"
sshd_enable=\"YES\"
dumpdev=\"AUTO\"

ifconfig_em0=\"inet $1 netmask 255.255.255.255 broadcast $1\"
static_routes=\"ominiv4\"
route_ominiv4=\"-net $2/32 -iface em0\"
defaultrouter=\"$2\"" >> /etc/rc.conf

rm -f /etc/resolv.conf
echo "nameserver 8.8.8.8" >> /etc/resolv.conf
echo "nameserver 8.8.4.4" >> /etc/resolv.conf

passwd root<<EOF
$3
$3
EOF

reboot
